(DEF 'directional-verb '(LAMBDA ()
  (SETQ R (random 5))
  (PRINC
  (COND
   ((EQ R 1) "puke")
   ((EQ R 2) "vomit")
   ((EQ R 3) "fart")
   ((EQ R 4) "piss")
   ((EQ R 5) "shout obscenities")))))

(DEF 'direction '(LAMBDA ()
  (SETQ R (random 3))
  (PRINC
  (COND
   ((EQ R 1) " at you")
   ((EQ R 2) " in your face")
   ((EQ R 2) " in your general direction")))))

(DEF 'verb-phrase '(LAMBDA ()
  (SETQ R (random 5))
    (COND
     ((EQ R 1) (directional-verb)  (direction))
     ((EQ R 2) (PRINC "cry"))
